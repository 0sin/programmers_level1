  function solution(s) {
    var answer = s;
    var num = s.length;
    
    if (num % 2 == 0) {
        var num1 = num / 2 - 1;
        var num2 = num / 2;
        return answer[num1] + answer[num2];
    } else {
        num = num / 2 - 0.5;
        return answer[num];
    }
  }
